## Quick Start

```typescript
import { generateText, createGateway } from 'ai';

const gateway = createGateway({ apiKey: process.env.AI_GATEWAY_API_KEY });

const { text } = await generateText({
  model: gateway('openai/gpt-5-chat'),
  prompt: 'Explain machine learning',
});
```

## Capabilities

### Vision (Image Input)

```typescript
const { text } = await generateText({
  model: gateway('openai/gpt-4o'),
  messages: [{
    role: 'user',
    content: [
      { type: 'text', text: 'Describe this image' },
      { type: 'image', image: imageUrl },
    ],
  }],
});
```

### Tool Calling

```typescript
const { toolCalls } = await generateText({
  model: gateway('openai/gpt-5-chat'),
  prompt: 'Calculate 15 + 27',
  tools: {
    calculator: {
      description: 'Perform math operations',
      inputSchema: jsonSchema<{ expression: string }>({
        type: 'object',
        properties: { expression: { type: 'string' } },
        required: ['expression'],
      }),
      execute: async ({ expression }) => eval(expression),
    },
  },
});
```

### Codex Models (Code-Optimized)

GPT-5 Codex and GPT-5.1 Codex models are optimized for code generation:

```typescript
// Use tools/JSON, not plain text prompts
const { toolCalls } = await generateText({
  model: gateway('openai/gpt-5.1-codex'),
  prompt: 'Create a React component for a todo list',
  tools: {
    writeCode: {
      description: 'Write code to a file',
      inputSchema: jsonSchema<{ filename: string; code: string }>({
        type: 'object',
        properties: {
          filename: { type: 'string' },
          code: { type: 'string' },
        },
        required: ['filename', 'code'],
      }),
      execute: async ({ filename, code }) => ({ written: filename }),
    },
  },
});
```

### Reasoning Model (GPT-5.1 Thinking)

```typescript
const { text, reasoning } = await generateText({
  model: gateway('openai/gpt-5.1-thinking'),
  prompt: 'Solve this complex problem...',
  providerOptions: {
    openai: {
      reasoningSummary: 'detailed',
    },
  },
});
```
