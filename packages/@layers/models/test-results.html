How to Use Image Generation

import { experimental_generateImage as generateImage, generateText, createGateway } from 'ai';

const gateway = createGateway({ apiKey: process.env.AI_GATEWAY_API_KEY });

// ===== BFL FLUX Models (Flux 2.x and 1.x) =====
const fluxResult = await generateImage({
  model: 'bfl/flux-2-pro',  // or flux-2-flex, flux-2-klein-4b, flux-2-klein-9b
  prompt: 'A futuristic cityscape at sunset',
  aspectRatio: '16:9',  // supports custom aspect ratios
});
console.log(`Generated ${fluxResult.images.length} image(s)`);
// Access: fluxResult.images[0].base64 or .uint8Array

// ===== Google Imagen Models =====
const imagenResult = await generateImage({
  model: 'google/imagen-4.0-fast-generate-001',  // or imagen-4.0-ultra-generate-001
  prompt: 'A serene mountain landscape',
  aspectRatio: '1:1',
  n: 2,  // generate multiple images
});
console.log(`Generated ${imagenResult.images.length} image(s)`);

// ===== Gemini Multimodal (Text + Image) =====
// These models can generate BOTH text and images
const geminiResult = await generateText({
  model: gateway('google/gemini-2.5-flash-image'),  // or gemini-3-pro-image
  prompt: 'Generate an image of a red circle and describe it',
});
console.log('Text:', geminiResult.text);
// Images returned in result.files[]
for (const file of geminiResult.files || []) {
  if (file.mediaType?.startsWith('image/')) {
    fs.writeFileSync('output.png', file.uint8Array);
  }
}
