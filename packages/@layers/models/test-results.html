How to Use Image Generation

// Image-only models (Flux, Imagen)
import { experimental_generateImage as generateImage } from 'ai';

const { images } = await generateImage({
  model: 'bfl/flux-pro-1.1',  // Just pass string to gateway
  prompt: 'A futuristic cityscape at sunset',
  aspectRatio: '16:9',
});

// Multimodal models (Gemini Image)
import { generateText } from 'ai';

const result = await generateText({
  model: gateway('google/gemini-2.5-flash-image'),
  prompt: 'Generate an image of a red circle',
});

// Images are in result.files[]
for (const file of result.files) {
  if (file.mediaType?.startsWith('image/')) {
    fs.writeFileSync('output.png', file.uint8Array);
  }
}

Anthropic Claude - Extended Thinking

Code Example

import { generateText } from 'ai';

const result = await generateText({
  model: gateway('anthropic/claude-sonnet-4.5'),
  prompt: 'Think step by step: What is 15 + 27?',
  maxOutputTokens: 500,
  providerOptions: {
    anthropic: {
      thinking: { type: 'enabled', budgetTokens: 1024 }
    }
  }
});

// Reasoning is an ARRAY of thinking blocks
console.log('Reasoning:', result.reasoning);
// → [{ type: 'thinking', thinking: 'Let me add these...' }]

// ReasoningText is the concatenated string
console.log('Reasoning Text:', result.reasoningText);
// → "Let me add these numbers step by step..."

console.log('Final Answer:', result.text);
// → "42"

Sample Response Structure

{
  text: "The answer is 42.",
  reasoning: [
    { type: "thinking", thinking: "I need to add 15 and 27..." },
    { type: "thinking", thinking: "15 + 27 = 42" }
  ],
  reasoningText: "I need to add 15 and 27...\n15 + 27 = 42"
}

OpenAI GPT-5.1 Thinking

Code Example

import { generateText } from 'ai';

const result = await generateText({
  model: gateway('openai/gpt-5.1-thinking'),
  prompt: 'Think step by step: What is 15 + 27?',
  maxOutputTokens: 500,
});

// OpenAI embeds reasoning directly in the text response
console.log('Response:', result.text);
// → "Let me think through this step by step:\n\n1. Start with 15\n2. Add 27\n3. 15 + 27 = 42\n\nThe answer is 42."

// May also have reasoning object (varies by response)
if (result.reasoning) {
  console.log('Reasoning type:', typeof result.reasoning);
  // Could be array or object - use JSON.stringify
  console.log('Reasoning:', JSON.stringify(result.reasoning).substring(0, 300));
}

Sample Response Structure

{
  text: "Let me think through this step by step:\n\n1. Start with 15\n2. Add 27\n3. 15 + 27 = 42\n\nThe answer is 42.",
  reasoning: [...] // May be present, format varies
}

Perplexity Sonar Reasoning Pro

Code Example

import { generateText } from 'ai';

const result = await generateText({
  model: gateway('perplexity/sonar-reasoning-pro'),
  prompt: 'Think step by step: What is 15 + 27?',
  maxOutputTokens: 500,
});

// Perplexity is a reasoning-ONLY model
// All output includes step-by-step thinking in the text
console.log('Response:', result.text);
// → "To solve 15 + 27:\n\nStep 1: Break down the numbers...\nStep 2: Add the ones place...\n\nFinal answer: 42"

Sample Response Structure

{
  text: "To solve 15 + 27:\n\nStep 1: I'll break down the addition...\nStep 2: 15 + 20 = 35...\nStep 3: 35 + 7 = 42\n\nThe answer is **42**."
  // No separate reasoning field - it's a reasoning-only model
}

⚠️ This model has web search built-in. Best for research questions requiring real-time information + reasoning.
